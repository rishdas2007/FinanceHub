    get() {
      return this.pool.pop() || this.createNew();
    }

    release(obj: any) {
      if (this.pool.length < this.maxSize) {
        this.reset(obj);
        this.pool.push(obj);
      }
    }

    private createNew() {
      return { data: [], meta: {} };
    }

    private reset(obj: any) {
      obj.data = [];
      obj.meta = {};
    }
  }
